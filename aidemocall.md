# services/presentation_ai.py
class PresentationAI:
    def __init__(self, meeting_room_url: str, customer_id: str):
        self.daily_client = DailyClient()
        self.room = meeting_room_url
        self.customer = get_customer(customer_id)
        
        # Load presentation script
        self.script = PresentationScript()
        self.current_phase = "warmup"
        
        # Audio streaming
        self.openai_realtime = OpenAIRealtimeAPI()
        
    async def join_meeting(self):
        """AI joins Daily.co meeting room"""
        await self.daily_client.join_room(
            room_url=self.room,
            display_name="Claude - Warlord AI"
        )
        
        # Wait for human team member to join
        await self.wait_for_human_participant()
        
        # Start presentation
        await self.start_phase_1_warmup()
    
    async def start_phase_1_warmup(self):
        """Phase 1: Conversational introduction (streaming)"""
        
        system_prompt = self.script.get_phase_1_prompt(
            customer_name=self.customer.name,
            customer_company=self.customer.company
        )
        
        # Start streaming audio
        await self.openai_realtime.start_stream(
            system_prompt=system_prompt,
            voice="alloy",  # Or custom voice
            on_interruption=self.handle_interruption
        )
        
        # AI speaks opening script
        await self.speak(self.script.phase_1_opening)
        
        # Wait for customer responses, handle dynamically
        while self.current_phase == "warmup":
            user_speech = await self.listen_for_speech()
            
            # AI responds naturally
            response = await self.openai_realtime.respond(user_speech)
            await self.speak(response)
            
            # Check if it's time to transition
            if self.script.should_transition_to_ppt():
                await self.transition_to_ppt_mode()
                break
    
    async def transition_to_ppt_mode(self):
        """Switch from streaming to pre-scripted PPT presentation"""
        
        # Signal to screen share
        await self.daily_client.start_screen_share(
            ppt_url="https://cdn.frontofai.com/presentations/sales-deck.pdf"
        )
        
        # Stop streaming, switch to narration mode
        await self.openai_realtime.stop_stream()
        
        self.current_phase = "ppt_narration"
        await self.start_phase_2_transformation()
    
    async def start_phase_2_transformation(self):
        """Phase 2: Problem + Old Solutions (PPT with narration)"""
        
        slides = self.script.phase_2_slides
        
        for slide in slides:
            # Show slide
            await self.show_slide(slide.number)
            
            # AI narrates slide (pre-recorded or TTS)
            await self.play_narration(slide.audio_file)
            
            # Wait for narration to finish + pause
            await asyncio.sleep(slide.duration + slide.pause)
        
        await self.start_phase_3_reveal()
    
    async def start_phase_3_reveal(self):
        """Phase 3: Solution reveal (Jobs-style 'one more thing')"""
        
        # Slide 4: THE SOLUTION
        await self.show_slide(4)
        await self.play_narration_with_pauses(
            audio=self.script.slide_4_audio,
            pause_points=[3.0, 6.0]  # Strategic pauses
        )
        
        # Slide 5: HOW IT WORKS + DEMO AUDIO
        await self.show_slide(5)
        await self.play_narration(self.script.slide_5_audio)
        
        # Play real call demo
        await self.play_demo_call(self.script.demo_audio_url)
        
        # Slide 6: THE DIFFERENCE
        await self.show_slide(6)
        await self.play_narration(self.script.slide_6_audio)
        
        # Slide 7: ONE MORE THING (killer feature)
        await self.show_slide(7)
        await self.play_narration_with_pauses(
            audio=self.script.slide_7_audio,
            pause_points=[2.0]  # Pause after "one more thing"
        )
        
        # Slide 8: REAL RESULTS
        await self.show_slide(8)
        await self.play_narration(self.script.slide_8_audio)
        
        # Transition to human
        await self.handoff_to_human()
    
    async def handoff_to_human(self):
        """AI hands control to human team member"""
        
        # Stop screen share
        await self.daily_client.stop_screen_share()
        
        # AI speaks transition
        await self.speak(
            "Thanks everyone. My colleague is going to take it from "
            "here to answer any questions you have."
        )
        
        # Mute AI, unmute human
        await self.daily_client.mute_self()
        
        # Signal to human (via backend event)
        await self.notify_human_to_take_over()
        
        self.current_phase = "human_close"
    
    async def handle_interruption(self, user_speech: str):
        """Handle customer interruptions during presentation"""

            f"Customer interrupted: {user_speech}. Acknowledge briefly, "
            f"then return to presentation."
        )
        
        await self.speak(response)
        
        # Resume narration
        await self.resume_narration()

class AIAvatarPresenter {

    this.didClient = null;
    this.avatarStream = null;
  }
  
  async joinMeeting(roomUrl) {

      videoSource: false  // Will use avatar stream
    });
    
    await this.dailyCall.join({ url: roomUrl });
    
    // 2. Initialize D-ID avatar
    this.didClient = new DID({
      auth: { apiKey: process.env.DID_API_KEY }
    });
    
    const avatarSession = await this.didClient.agents.create({
      presenter_id: 'amy-jcwCkr1grs',
      driver_id: 'uM00QMwJ9x',
      
      background: {
        type: 'image',
        url: 'https://cdn.frontofai.com/backgrounds/tech-office.jpg'

      text: text
    });
  }
  
  async destroy() {
    if (this.avatarStream) {
      await this.didClient.agents.destroy(this.avatarStream.sessionId);
    }
    if (this.dailyCall) {
      await this.dailyCall.leave();
      await this.dailyCall.destroy();
    }
  }
}
